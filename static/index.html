API Endpoints

  • POST /download
    Download any Instagram media (auto-detect type)
  • POST /download/reel
    Download Instagram Reel
  • POST /download/story
    Download Instagram Story
  • POST /download/post
    Download Instagram Post
  • POST /download/igtv
    Download Instagram IGTV
  • POST /info
    Get media information without downloading
  • GET /health
    Health check endpoint
  • GET /proxy?url=MEDIA_URL
    Direct download through proxy

API Usage Examples

cURL

curl -X POST http://localhost:5000/download \
-H "Content-Type: application/json" \
-d '{"url": "https://www.instagram.com/reel/example123/"}'

Python

import requests

api_url = "http://localhost:5000/download/reel"
data = {"url": "https://www.instagram.com/reel/example123/"}

response = requests.post(api_url, json=data)
result = response.json()

if result['status'] == 'success':
    for media in result['media']:
        print(f"Download URL: {media['url']}")

JavaScript (Fetch)

const downloadInstagram = async (url) => {
  const response = await fetch('http://localhost:5000/download', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({ url })
  });
  return await response.json();
};

// Usage
downloadInstagram('https://www.instagram.com/reel/example123/')
  .then(data => console.log(data));

Response Format

Success Response

{
  "status": "success",
  "media": [
    {
      "url": "https://example.com/video.mp4",
      "type": "video",
      "quality": "hd",
      "size": "15.2 MB"
    }
  ],
  "metadata": {
    "title": "Instagram Reel"
  },
  "count": 1
}

Error Response

{
  "status": "error",
  "message": "Reel not found or private"
}
